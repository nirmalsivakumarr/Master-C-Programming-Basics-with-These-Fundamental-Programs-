#include <stdio.h>

int main() {
    int arr[100],n,i,min;
    printf("Enter number of elements : ");
    scanf("%d",&n);
    printf("Enter %d elements:\n");
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
min=arr[0];
for(i=1;i<n;i++){
    if(arr[i]<min){
        min=arr[i];
    }
}
printf("Minimum Element = %d ", min);
    
    return 0;
}

OUTPUT:

Enter number of elements : 4
Enter 4 elements:
1 2 3 4
Minimum Element = 1
