#include <stdio.h>

int main() {
    int arr[100],n,i;
    printf("Enter number of elements : ");
    scanf("%d",&n);
    printf("Enter %d elements:\n");
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
printf("Array in reverse order : \n");
for(i=n-1;i>=0;i--){
    printf("%d ",arr[i]);
    
    }
printf("\n");
    
    return 0;
}

OUTPUT: 
Enter number of elements : 3
Enter 3 elements:
1 2 3
Array in reverse order : 
3 2 1
