#include <stdio.h>

int main() {
    int arr[100],n,i,max;
    printf("Enter number of elements : ");
    scanf("%d",&n);
    printf("Enter %d elements:\n");
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
max=arr[0];
for(i=1;i<n;i++){
    if(arr[i]>max){
        max=arr[i];
    }
}
printf("Maximum Element = %d ", max);
    
    return 0;
}

OUTPUT:

Enter number of elements : 3
Enter 3 elements:
1 2 3
Maximum Element = 3
