#include <stdio.h>
#include <ctype.h>
int main() {
    char str[200];
    int vowels=0,consonants=0,i;
    printf("Enter the string:");
    scanf("%[^\n]",str);
    for(i=0;str[i]!=0;i++)
    {
        char ch = tolower(str[i]);
        if(ch >= 'a' && ch <='z'){
            if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u')
                vowels++;
            else
                consonants++;
        }
    }
    printf("Vowels:%d\n",vowels);
    printf("Consonants: %d\n",consonants);
    return 0;
}
