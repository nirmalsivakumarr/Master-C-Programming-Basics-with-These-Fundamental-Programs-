#include <stdio.h>

int main() {
    int arr[100],n,i,x,count=0;
    printf("Enter number of elements : ");
    scanf("%d",&n);
    printf("Enter %d elements:\n");
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
printf("Enter number to count: ");
scanf("%d",&x);
    for(i=0;i<n;i++){
        if(arr[i]==x)
        count++;
    }
    printf("%d occurs %d times\n",x,count);
    
    return 0;
}

OUTPUT:

Enter number of elements : 5
Enter 5 elements:
1 2 3 2 1
Enter number to count: 2
2 occurs 2 times
