#include <stdio.h>

int main() {
    int x = 2, y = 2, z = 2;
    int a[2][2][2] = {{{1, 2}, {3, 4}},{{5, 6}, {7, 8}}};
    int b[2][2][2] = {{{8, 7}, {6, 5}},{{4, 3}, {2, 1}}};
    int sum[2][2][2];
    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            for (int k = 0; k < z; k++) {
                sum[i][j][k] = a[i][j][k] + b[i][j][k];
            }
        }
    }
    printf("Sum of the two 3D arrays:\n");
    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            for (int k = 0; k < z; k++) {
                printf("sum[%d][%d][%d] = %d\n", i, j, k, sum[i][j][k]);
            }
        }
    }

    return 0;
}
