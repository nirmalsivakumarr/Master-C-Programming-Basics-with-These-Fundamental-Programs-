#include <stdio.h>

int main() {
    int x = 2, y = 2, z = 2; 
    int a[2][2][2] = {{{1, 2}, {3, 4}},{{5, 6}, {7, 8}}};
    int b[2][2][2] = {{{8, 7}, {6, 5}},{{4, 3}, {2, 1}}};
    int product[2][2][2];
    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            for (int k = 0; k < z; k++) {
                product[i][j][k] = a[i][j][k] * b[i][j][k];
            }
        }
    }
    printf("Element-wise multiplication result:\n");
    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            for (int k = 0; k < z; k++) {
                printf("product[%d][%d][%d] = %d\n", i, j, k, product[i][j][k]);
            }
        }
    }

    return 0;
}
